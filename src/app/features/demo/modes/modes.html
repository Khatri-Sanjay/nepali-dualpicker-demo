Selection Modes

Single date, date range, or multiple dates

Single Date

config: DatePickerConfig = {
  mode: 'single',
  placeholder: 'Pick a Single Date'
};

Date Range

config: DatePickerConfig = {
  mode: 'range',
  closeOnSelect: false,
  placeholder: 'Select Date Range'
};

Multiple Dates (Max 5)

config: DatePickerConfig = {
  mode: 'multiple',
  closeOnSelect: false,
  multipleSelectionSettings: {
    maxSelections: 5
  }
};
